%\newcounter{itemcount}
%\newcounter{finalcount}
%\newcommand{\rcite}[1]{%
	%\setcounter{itemcount}{0}%
	%\renewcommand*{\do}[1]{\stepcounter{itemcount}\setcounter{finalcount}{\value{itemcount}}}%
  %\docsvlist{#1}%
	%\setcounter{itemcount}{0}%
	%\renewcommand*{\do}[1]{\stepcounter{itemcount}%
		%\ifnumcomp{\value{itemcount}}{=}{1}{%
				%R-\ref{##1}%
		%}{%
			%\ifnumcomp{\value{finalcount}}{>}{3}{%
				%\ifnumcomp{\value{itemcount}}{=}{\value{finalcount}}{ -- R-\ref{##1}}{}%
			%}{, R-\ref{##1}}%
		%}%
	%}%
  %\docsvlist{#1}%
%}
