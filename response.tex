\newcounter{itemcount}
\newcommand{\rcite}[1]{%
	\def\rlist{}%
	\def\citeform##1{\docprefix-##1}%
	\setcounter{itemcount}{0}%
  	\renewcommand*{\do}[1]{\stepcounter{itemcount}%
      \ifnumcomp{\value{itemcount}}{=}{1}{}{\appto\rlist{, }}%
      \appto\rlist{\docprefix-##1}%
    }% How to process each item
  	\docsvlist{#1}% Process list
	\cite{\rlist}%
	\def\citeform##1{##1}%
}

	\nocitelref{#1}%
	\rcite{#1}%

\newcounter{eqcounter}
\newcommand{\revision}[1]{
  \setcounter{eqcounter}{\value{equation}}%
  \renewcommand{\theequation}{\arabic{equation}}%
  \par%
  \begin{mdframed}[userdefinedwidth=0.95\linewidth,align=center]%
    \ExecuteMetaData[\maintext]{#1}%
  \end{mdframed}%
  \setcounter{equation}{\value{eqcounter}}%
  \renewcommand{\theequation}{\docprefix-\arabic{equation}}%
}

% Usage example: 
% If the tex that you transfer over using \revision contains a \label{thmname} for an environment (e.g., Theorem, Lemma, Definition), it will not have the same numbering as that in \maintext. 
% Call \xrcounterref{environment_name}{thmname} before \revision to fix this. 
\newcommand{\xrcounterref}[2]{
  \setcounter{#1}{\getrefnumber{#2}}
  \addtocounter{#1}{-1}
}

\noindent\textbf{Recommendation:} Major Revision

\begin{comment}
A lot of comments.
\end{comment}

\begin{response}
  Thank you for your comment. Please always use the formal ``thank you'' instead of ``thanks''. Please always start your response politely and in a conciliatory tone even if you are frustrated with the comment. We can be firm, friendly and respectful. After all, the reviewer has sacrificed his or her precious time to read our manuscript and provide comments, which he or she is not obligated to do.

  Using the xr package, I can reference \cref{sect:General} and \cref{eq:X} of tips.tex. To do this, you need to build response.tex in the same folder containing tips.aux.

  Next, I cite \cite{LenTayQui:J16,Str:93}. The references displayed are from tips.aux and tips.bbl thanks to xr. Finally, the catchfilebetweentags package allows me to quote the following from \cref{sect:Response} in tips.tex:
  \revision{tag:catch}

  Note that the tag name cannot include underscores. Another quote from tips.tex:
  \revision{tag:equation}
\end{response}

\begin{comment}
Do something.
\end{comment}
\begin{response}
  Thank you for your suggestion. I will now cite some references that appear only in the bibliography for this response using \lcite{AceDahLobOzd:11,ChePap:95,Cov:69}. The steps to be taken are:
  \begin{enumerate}[i)]
    \item latex response
    \item bibtex lref.aux
    \item latex response
    \item bibtex lref.aux
    \item\label{it:rename} copy lref.aux to lref-response.aux, and lref.bbl to lref-response.bbl
    \item latex response
    \item latex response
    \item latex response
  \end{enumerate}
  These steps are automated in the lcite\_.bat file. \textbf{If anyone can figure out a better solution, please let me know.}

\end{response}
